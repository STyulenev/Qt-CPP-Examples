cmake_minimum_required(VERSION 3.16)

project(QML-QQmlEngineExtensionPlugin-Example
    VERSION 0.1
    DESCRIPTION "Project with sample QQmlEngineExtensionPlugin"
    HOMEPAGE_URL "https://github.com/STyulenev/Qt-CPP-Examples"
    LANGUAGES CXX
)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(QT NAMES Qt6 Qt5 COMPONENTS Qml Quick REQUIRED)
find_package(Qt${QT_VERSION_MAJOR} COMPONENTS Qml Quick REQUIRED)

set(PROJECT_SOURCES
    main.cpp
    resource.qrc
)

if(${QT_VERSION_MAJOR} GREATER_EQUAL 6)
    qt_add_executable(${PROJECT_NAME}
        MANUAL_FINALIZATION
        ${PROJECT_SOURCES}
    )
else()
    add_executable(${PROJECT_NAME}
       ${PROJECT_SOURCES}
    )
endif()

list(APPEND QML_DIRS "${CMAKE_BINARY_DIR}/plugins/")
list(REMOVE_DUPLICATES QML_DIRS)
set(QML_IMPORT_PATH "${QML_DIRS}" CACHE STRING "QML import paths")

target_compile_definitions(${PROJECT_NAME} PRIVATE $<$<OR:$<CONFIG:Debug>,$<CONFIG:RelWithDebInfo>>:QT_QML_DEBUG>)
target_link_libraries(${PROJECT_NAME} PRIVATE
    Qt${QT_VERSION_MAJOR}::Qml
    Qt${QT_VERSION_MAJOR}::Quick
)

install(TARGETS ${PROJECT_NAME}
   RUNTIME DESTINATION "${CMAKE_BINARY_DIR}"
   BUNDLE DESTINATION "${CMAKE_BINARY_DIR}"
   LIBRARY DESTINATION "${CMAKE_BINARY_DIR}"
)

add_subdirectory(plugins)
